# Pin Order Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Let each user pin their own orders so pinned orders float to the top of Order Registry, persisted cross-device via SharePoint.

**Architecture:** A new `pinnedOrderService.ts` calls a Power Automate flow (`VITE_API_PINNED_ORDER_URL`) backed by the `App_UserPinnedOrders` SharePoint list. `OrderRegistry.tsx` loads the user's pinned IDs on mount, applies optimistic updates on toggle, and sorts pinned orders to the top. `PermissionContext` already exposes `userEmail` — no context changes needed.

**Tech Stack:** React 19, TypeScript, Vite 6, Tailwind CSS v4, Lucide React, Power Automate HTTP flow, SharePoint `App_UserPinnedOrders` list.

// src/services/pinnedOrderService.ts
const FLOW_URL = import.meta.env.VITE_API_PINNED_ORDER_URL as string;

interface PinRecord {
  OrderId: number;
}

async function call<T>(body: object): Promise<T> {
  const res = await fetch(FLOW_URL, {
    method: "POST",
    headers: { "Content-Type": "application/json" },
    body: JSON.stringify(body),
  });
  if (!res.ok) throw new Error(`pinnedOrderService: HTTP ${res.status}`);
  return res.json() as Promise<T>;
}

export const pinnedOrderService = {
  async getPinned(userEmail: string): Promise<number[]> {
    const data = await call<{ value: PinRecord[] }>({
      action: "GET_BY_USER",
      data: { UserEmail: userEmail },
      userEmail,
    });
    return (data.value ?? []).map((r) => Number(r.OrderId));
  },

  async pin(userEmail: string, orderId: number): Promise<void> {
    await call<void>({
      action: "PIN",
      data: { UserEmail: userEmail, OrderId: orderId },
      userEmail,
    });
  },

  async unpin(userEmail: string, orderId: number): Promise<void> {
    await call<void>({
      action: "UNPIN",
      data: { UserEmail: userEmail, OrderId: orderId },
      userEmail,
    });
  },
};
```

const handlePinToggle = async (orderId: number) => {
  const isPinned = pinnedIds.has(orderId);
  const prev = new Set(pinnedIds);
  setPinnedIds((s) => {
    const next = new Set(s);
    if (isPinned) next.delete(orderId); else next.add(orderId);
    return next;
  });
  try {
    if (isPinned) {
      await pinnedOrderService.unpin(userEmail, orderId);
    } else {
      await pinnedOrderService.pin(userEmail, orderId);
    }
  } catch {
    setPinnedIds(prev);
  }
};
```

const filteredOrders = [...allOrders]
  .sort((a, b) => b.id - a.id)
  .filter((order) => {
    if (activeTab === "Pending") {
      if (["Completed", "Cancelled"].includes(order.Status)) return false;
    } else if (activeTab === "Completed") {
      if (order.Status !== "Completed") return false;
    }

    if (
      providerFilter !== "All" &&
      !(order.CloudProvider ?? "").includes(providerFilter)
    ) {
      return false;
    }

    if (statusFilter !== "All" && order.Status !== statusFilter) {
      return false;
    }

    if (searchQuery) {
      const query = searchQuery.toLowerCase();
      if (
        !(order.Title ?? "").toLowerCase().includes(query) &&
        !(order.CustomerName ?? "").toLowerCase().includes(query) &&
        !(order.AccountID && order.AccountID.toLowerCase().includes(query))
      ) {
        return false;
      }
    }

    return true;
  })
  .sort((a, b) => {
    const aPinned = pinnedIds.has(a.id) ? 1 : 0;
    const bPinned = pinnedIds.has(b.id) ? 1 : 0;
    return bPinned - aPinned;
  });
